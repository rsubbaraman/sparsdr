########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(
    ${Boost_INCLUDE_DIR}
    ${PYTHON_INCLUDE_DIRS}
)
link_directories(${Boost_LIBRARY_DIRS})

list(APPEND sparsdr_sources
    compressing_usrp_source_impl.cc
    compressing_pluto_source_impl.cc
    iio_device_source_impl.cc
    reconstruct_impl.cc
    endian_converter.cc
    swap_16_impl.cc
    combined_usrp_receiver_impl.cc
    combined_common.cc
    combined_pluto_receiver_impl.cc
)

set(sparsdr_sources "${sparsdr_sources}" PARENT_SCOPE)
if(NOT sparsdr_sources)
	MESSAGE(STATUS "No C++ sources... skipping lib/")
	return()
endif(NOT sparsdr_sources)

add_library(gnuradio-sparsdr SHARED ${sparsdr_sources})
target_link_libraries(gnuradio-sparsdr
	${Boost_LIBRARIES}
	${GNURADIO_ALL_LIBRARIES}
    ${PYTHON_LIBRARIES}
    ${IIO_LIBRARIES}
    ${UHD_LIBRARIES}
)
set_target_properties(gnuradio-sparsdr PROPERTIES DEFINE_SYMBOL "gnuradio_sparsdr_EXPORTS")
target_include_directories(gnuradio-sparsdr
    PRIVATE
    ${IIO_INCLUDE_DIRS}
)
# Extra compiler warnings
target_compile_options(gnuradio-sparsdr PRIVATE -Wall -Wextra -pedantic)

if(APPLE)
    set_target_properties(gnuradio-sparsdr PROPERTIES
        INSTALL_NAME_DIR "${CMAKE_INSTALL_PREFIX}/lib"
    )
endif(APPLE)

########################################################################
# Install built library files
########################################################################
include(GrMiscUtils)
GR_LIBRARY_FOO(gnuradio-sparsdr RUNTIME_COMPONENT "sparsdr_runtime" DEVEL_COMPONENT "sparsdr_devel")

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_sparsdr_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_sparsdr.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_sparsdr.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_swap_16.cc
)

add_executable(test-sparsdr ${test_sparsdr_sources})

target_link_libraries(
  test-sparsdr
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-sparsdr
)

GR_ADD_TEST(test_sparsdr test-sparsdr)

########################################################################
# Print summary
########################################################################
message(STATUS "Using install prefix: ${CMAKE_INSTALL_PREFIX}")
message(STATUS "Building for version: ${VERSION} / ${LIBVER}")
